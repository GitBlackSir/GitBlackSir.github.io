分类

类别 平均情况 —最好情况 —最坏情况 —辅助空间 —稳定性
冒泡排序 O(n^2) O(n) O(n^2) O(1) 稳定

基数排序的复杂度中，r代表关键字的基数，d代表长度，n代表关键字的个数。

冒泡排序

时间复杂度 O(n^2) 左右比较，若左大于右,大的放后面（交换） 第一趟排序后最小的在前面 第二趟排序范围1到n-1，第二小的在下标1的位置 ….

void solution(int a[],int len){
    int lastSwapPos = 0,lastSwapPosTemp = 0;
    for (int i = 0; i < len - 1; i++){
        lastSwapPos = lastSwapPosTemp;
        for (int j = len - 1; j >lastSwapPos; j--){
            if (a[j - 1] > a[j]){
                int temp = a[j - 1];
                a[j - 1] = a[j];
                a[j] = temp;
                lastSwapPosTemp = j;
            }
        }
        if (lastSwapPos == lastSwapPosTemp)
            break;
}

插入排序

时间复杂度O(n^2) 位置1和位置0比较 如果位置1的小，交换 位置2和位置1比较 如果位置2的小，交换 位置1和位置0比较 如果位置1的小，交换 ….

void solution(int a[],int len) {
        int i, j;
        for(i=1;i<len;i++) {
           if(a[i]<a[i-1]) {
            int temp = a[i];
             for(j=i-1;a[j]>temp;j--)
             		a[j+1] = a[j];
             a[j+1] = temp;
           }
        }
}

选择排序

时间复杂度 O(n^2) 在整个数组中从0-n-1 选出最小的放到0位置 接着从数组中从 1-n-1 选出最小的放到1位置 ….

void solution(int a[], int len) {
    int min = 0;
    for (int i = 0; i < len; ++i) {
        min = i;
        for (int j = i+1; j <len ; ++j) {
            if(a[min] > a[j])
                min = j;
        }
        if(min != i)
            swap(a[i],a[min]);
    }
}

快速排序

选择一个基准元素（随机选一个数或第一个或最后一个） 一趟排序将序列分割成独立的两部分（一部分比基准元素小，一部分比基准元素大） 此时基准元素在在排序后的正确位置 然后分别对两部分用同样的方法进行排序，直到整个序列有序

void qsort(int a[],int left,int right){
    if(left >= right)
        return;
    int first = left;
    int last  = right;
    int key   = a[first];

    while (first < last) {
        while (first < last && a[last] > key)
            --last;
        a[first] = a[last];
        while (first < last && a[first] <= key)
            ++first;
        a[last] = a[first];
    }
    a[first] = key;
    qsort(a,left,first-1);
    qsort(a,first+1,right);
}
