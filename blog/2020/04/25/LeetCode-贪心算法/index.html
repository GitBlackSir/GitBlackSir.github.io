class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {
        int fives = 0, tens = 0;
        for(auto b : bills)
        {
            if(b == 5) fives++;
            else if(b == 10)
            {
                tens++;
                if(fives) fives--;
                else return false;
            }
            else
            {
                int t = 15;
                if(tens) { tens--; t -= 10; }
                while(fives && t) { fives--; t -= 5;}
                if(t) return false;
            }
        }
        return true;
    }
};

class Solution {
public:
    int wiggleMaxLength(vector<int>& f) {
        f.erase(unique(f.begin(), f.end()), f.end());
        if(f.size() <= 2) return f.size();
        int res = 2;
        for(int i = 1; i + 1 < f.size(); i++)
        {
            int a = f[i-1], b = f[i], c = f[i+1];
            if(a < b && b > c) res++;
            else if(a > b && b < c) res++;
        }
        return res;
    }
};

class Solution {
public:
    bool canJump(vector<int>& nums) {
        int dist = 0;
        for(int i = 0; i < nums.size() && i <= dist; i++)
            dist = max(dist, nums[i] + i);
        return dist >= nums.size() - 1;
    }
};

class Solution {
public:
    int jump(vector<int>& nums) {
        if(nums.size() == 1) return 0;
        int l = 0, r = 0, step = 0;
        while(l <= r)
        {
            int max_r = 0;
            for(int i = l; i <= r; i++)
                max_r = max(max_r, i + nums[i]);
            
            step++;
            l = r + 1;
            r = max_r;
            if(max_r >= nums.size() - 1) break;
        }
        return step;
    }
};

class Solution {
public:
    int canCompleteCircuit(vector<int>& g, vector<int>& c) {
        int n = g.size();
        for(int i = 0, j; i < n; i += j + 1)
        {
            int gas = 0;
            for(j = 0; j < n; j++)
            {
                int k = (i + j) % n;
                gas += g[k] - c[k];
                if(gas < 0)
                    break;
            }
            if(j == n)
                return i;
        }
        return -1;
    }
