class Solution {
public:
    int longestValidParentheses(string s) {
        int res = work(s);
        reverse(s.begin(), s.end());
        for(auto &c : s) c ^= 1;
        return max(res, work(s));
    }
    
    int work(string s){
        int res = 0;
        for(int i = 0, start = 0, count = 0; i < s.size(); i++){
            if(s[i] == '(') 
                count++;
            else{
                count--;
                if(count < 0) start = i + 1, count =0;
                else if (count == 0) res = max(res, i - start + 1);
            }
        }
        return res;
    }
};
