class Solution {
    public ListNode reverseBetween(ListNode head, int m, int n) {
        if(m == n) return head;
        var node = new ListNode(-1);
        node.next = head;
        var left = node;
        var right = node;
        
        for(int i = 0; i < m - 1; i++) left = left.next;
        for(int i = 0; i < n; i++) right = right.next;
        
        var leftTemp = left.next;
        var rightTemp = right.next;
        
        ListNode pre = null;
        var cur = leftTemp;
        while(cur != rightTemp){
            var next = cur.next;
            cur.next = pre;
            pre = cur;
            cur = next;
        }
        
        leftTemp.next = rightTemp;
        left.next = pre;
        return node.next;
    }
}

class Solution {
    private int len = 0;
    public ListNode rotateRight(ListNode head, int k) {
        if(head == null) return null;
        ListNode fir = head, sec = head;
        for(var i = head; i !=null; i = i.next) len++;
        
        k %= len;
        
        while(k-- > 0) fir = fir.next;
        
        while(fir.next != null){
            fir = fir.next;
            sec = sec.next;
        }
        
        fir.next = head;
        head = sec.next;
        sec.next = null;
        
        return head;
    }
}

class Solution {
    public ListNode sortList(ListNode head) {
        Queue<Integer> q = new PriorityQueue<>();
        var cur = head;
        while(cur != null){
            q.add(cur.val);
            cur = cur.next;
        }
        cur = head;
        while(cur != null){
            cur.val = q.poll();
            cur = cur.next;
        }
        return head;
    }
}
